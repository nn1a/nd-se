### 2. State Management

We use Zustand for simple, effective state management:

<CodeBlock title="Store Example" language="typescript">
{`import { create } from 'zustand'

interface UserState {
  user: User | null
  setUser: (user: User | null) => void
  isAuthenticated: boolean
}

export const useUserStore = create<UserState>((set) => ({
  user: null,
  setUser: (user) => set({ user, isAuthenticated: !!user }),
  isAuthenticated: false,
}))`}
</CodeBlock>

### 3. Data Fetching

React Query handles all server state:

<CodeBlock title="Custom Hook" language="typescript">
{`import { useQuery } from '@tanstack/react-query'
import { apiClient } from '@/lib/api'

export function useDocuments() {
  return useQuery({
    queryKey: ['documents'],
    queryFn: () => apiClient.get('/api/docs'),
    staleTime: 5 * 60 * 1000, // 5 minutes
  })
}

export function useDocument(slug: string) {
  return useQuery({
    queryKey: ['document', slug],
    queryFn: () => apiClient.get(\`/api/docs/\${slug}\`),
    enabled: !!slug,
  })
}`}
</CodeBlock>

## Common Patterns

### Error Boundaries

```tsx
'use client'

export function ErrorBoundary({ children }) {
  return (
    <ErrorBoundary
      fallback={<div>Something went wrong!</div>}
      onError={(error) => console.error('Error caught:', error)}
    >
      {children}
    </ErrorBoundary>
  )
}
```

### Loading States

```tsx
export function DocumentList() {
  const { data, isLoading, error } = useDocuments()
  
  if (isLoading) return <LoadingSpinner />
  if (error) return <ErrorMessage error={error} />
  
  return (
    <div>
      {data?.map(doc => (
        <DocumentCard key={doc.id} document={doc} />
      ))}
    </div>
  )
}
```

## Best Practices

<Card title="Development Best Practices">

- Keep components small and focused
- Use custom hooks for complex logic

- Optimize images and assets

- Use semantic HTML elements
- Add proper ARIA labels
- Ensure keyboard navigation
- Test with screen readers
